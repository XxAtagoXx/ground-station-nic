// Load a GLTF model var loader = new THREE.GLTFLoader(); loader.load( 'model.gltf', function ( gltf ) { scene.add( gltf.scene ); // Get the model's mesh var mesh = gltf.scene.children[0]; // Set the initial orientation mesh.rotation.x = 0; mesh.rotation.y = 0; mesh.rotation.z = 0; // Render loop function render() { requestAnimationFrame( render ); renderer.render( scene, camera ); } render(); // Update the orientation using x, y, and z values function updateOrientation(x, y, z) { mesh.rotation.x = x; mesh.rotation.y = y; mesh.rotation.z = z; } });
